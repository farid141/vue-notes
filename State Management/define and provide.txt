merupakan tools built-in dari vue

# provide melalui app

const app = createApp({})

app.provide(/* key */, /* value */)

# provide melalui komponen
function updateLocation() {
  location.value = 'South Pole'
}

provide('location', {
  location,
  updateLocation
})


# consume data
const { location, updateLocation } = inject('location', 'defaultValue')


# symbol untuk mengatasi key bentrok

// keys.js
export const myInjectionKey = Symbol()

// in provider component
import { provide } from 'vue'
import { myInjectionKey } from './keys.js'

provide(myInjectionKey, {
  /* data to provide */
})

import { inject } from 'vue'
